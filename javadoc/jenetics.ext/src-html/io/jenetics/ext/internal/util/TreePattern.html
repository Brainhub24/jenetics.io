044 /**
045  * A compiled representation of a <em>tree</em> pattern. A tree pattern,
046  * specified as a parentheses string, must first be compiled into an instance of
047  * this class. The resulting pattern can then be used to create a
048  * {@code TreeMatcher} object that can match arbitrary trees against the tree
049  * pattern. All of the state involved in performing a match resides in the
050  * matcher, so many matchers can share the same pattern.
051  * <p>
052  * The string representation of a tree pattern is a parenthesis tree string,
053  * with a special wildcard syntax for arbitrary sub-trees. The sub-trees
054  * variables are put into angle brackets:
055  * <pre>{@code
056  *     add(<x>,0)
057  *     mul(1,<y>)
058  * }</pre>
059  *
060  * @see TreeRewriteRule
061  * @see Tree#toParenthesesString()
062  * @see TreeMatcher
063  *
064  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
065  * @version 4.4
066  * @since 4.4
067  */
068 public final class TreePattern {
069 
070     private final Tree<Decl, ?> _pattern;
071     private final SortedSet<String> _variables;
072 
073     private TreePattern(final Tree<Decl, ?> pattern) {
074         _pattern = requireNonNull(pattern);
075 
076         final SortedSet<String> variables = new TreeSet<>();
077         for (Tree<Decl, ?> n : pattern) {
078             if (n.getValue().isVar) {
079                 if (!n.isLeaf()) {
080                     throw new IllegalArgumentException(format(
081                         "Variable node '%s' is not a leaf: %s",
082                         n.getValue(), n.toParenthesesString()
083                     ));
084                 }
085 
086                 variables.add(n.getValue().value);
087             }
088         }
089 
090         _variables = Collections.unmodifiableSortedSet(variables);
091     }

096      *
097      * @return the variables, defined in this pattern
098      */
099     public SortedSet<String> variables() {
100         return _variables;

149     public <V> Optional<TreeMatchResult<V>> match(
150         final Tree<V, ?> tree,
151         final BiPredicate<? super V, ? super String> equals
152     ) {
153         final Map<String, Tree<V, ?>> vars = new HashMap<>();
154         final boolean matches = matches(tree, _pattern, vars, equals);
155 
156         return matches
157             ? Optional.of(TreeMatchResult.of(tree, unmodifiableMap(vars)))
158             : Optional.empty();
159     }

181      * Tests whether the given input tree matches {@code this} pattern, using
182      * the given {@code equals} predicate. For comparing the tree node values
183      * with the pattern, the node values are converted to strings (with the
184      * {@link Object#toString()} first.
185      *
186      * @param tree the tree to be matched
187      * @return {@code true} if the {@code tree} matches {@code this} pattern,
188      *         {@code false} otherwise
189      * @throws NullPointerException if one of the arguments is {@code null}

192         return matches(tree, TreePattern::equals);

195     private static <V> boolean matches(
196         final Tree<V, ?> node,
197         final Tree<Decl, ?> pattern,
198         final Map<String, Tree<V, ?>> vars,
199         final BiPredicate<? super V, ? super String> equals
200     ) {
201         final Decl decl = pattern.getValue();
202 
203         if (decl.isVar) {
204             final Tree<V, ?> tree = vars.get(decl.value);
205             if (tree == null) {
206                 vars.put(decl.value, node);
207                 return true;
208             }
209 
210             return tree.equals(node);
211         } else {
212             final String p = pattern.getValue().value;
213             final V v = node.getValue();
214 
215             if (equals.test(v, p)) {
216                 if (node.childCount() == pattern.childCount()) {
217                     for (int i = 0; i < node.childCount(); ++i) {
218                         final Tree<V, ?> cn = node.getChild(i);
219                         final Tree<Decl, ?> cp = pattern.getChild(i);
220 
221                         if (!matches(cn, cp, vars, equals)) {
222                             return false;
223                         }
224                     }
225                     return true;
226                 } else {
227                     return false;
228                 }
229             } else {
230                 return false;
231             }
232         }
233     }

235     /**
236      * Expands {@code this} pattern with the given variable mapping and using
237      * the given value {@code mapper}. Missing {@code variables} mapping are
238      * removed from the expanded tree.
239      *
240      * @param variables the variables to use for expanding {@code this} pattern
241      * @param mapper the string value mapper
242      * @param <V> the tree node type
243      * @return the expanded tree pattern
244      * @throws NullPointerException if one of the arguments is {@code null}
245      * @throws IllegalArgumentException if not all needed variables are part
246      *         of the {@code variables} map
247      */
248     public <V> TreeNode<V> expand(
249         final Map<String, Tree<V, ?>> variables,
250         final Function<? super String, ? extends V> mapper
251     ) {
252         return expand(_pattern, variables, mapper);
253     }
254 
255     private static <V> TreeNode<V> expand(
256         final Tree<Decl, ?> template,
257         final Map<String, Tree<V, ?>> vars,
258         final Function<? super String, ? extends V> mapper
259     ) {
260         final Map<Path, Decl> paths = template.stream()
261             .filter((Tree<Decl, ?> n) -> n.getValue().isVar)
262             .collect(Collectors.toMap(t -> t.childPath(), t -> t.getValue()));
263 
264         //final Function<Decl, String> m = d -> d.isVar ? null: d.value;
265         final Function<Decl, String> m = d -> d.value;
266         final TreeNode<V> tree = TreeNode.ofTree(template, m.andThen(mapper));
267 
268         for (Map.Entry<Path, Decl> var : paths.entrySet()) {
269             final Path path = var.getKey();
270             final Decl decl = var.getValue();
271             final TreeNode<V> child = tree.childAtPath(path)
272                 .orElseThrow(AssertionError::new);
273 
274             final Tree<V, ?> replacement = vars.get(decl.value);
275             if (replacement != null) {
276                 tree.replaceAtPath(path, TreeNode.ofTree(replacement));
277             } else {
278                 tree.removeAtPath(path);
279             }
280         }
281 
282         return tree;
283     }
