030/**
031 * This class implements a linear congruential PRNG with additional bit-shift
032 * transition.
033 *
034 * <em>
035 * This is a port of the
036 * <a href="https://github.com/rabauke/trng4/blob/master/src/lcg64_shift.hpp">
037 * trng::lcg64_shift</a> PRNG class of the
038 * <a href="http://numbercrunch.de/trng/">TRNG</a> library created by Heiko
039 * Bauke.</em>
040 *
041 * <p>
042 * The following listing shows the actual PRNG implementation.
043 * <pre>{@code
044 * final long a, b = <params>
045 * long r = <seed>
046 *
047 * long nextLong() {
048 *     r = q*r + b;
049 *
050 *     long t = r;
051 *     t ^= t >>> 17;
052 *     t ^= t << 31;
053 *     t ^= t >>> 8;
054 *     return t;
055 * }
056 * }</pre>
057 *
058 * <p>
059 * <strong>Not that the base implementation of the {@code LCG64ShiftRandom}
060 * class is not thread-safe.</strong> If multiple threads requests random
061 * numbers from this class, it <i>must</i> be synchronized externally.
062 * Alternatively you can use the thread-safe implementations
063 * {@link LCG64ShiftRandom.ThreadSafe} or {@link LCG64ShiftRandom.ThreadLocal}.
064 *
065 * @see <a href="http://numbercrunch.de/trng/">TRNG</a>
066 *
067 * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
068 * @since 1.0
069 * @version 1.0
070 */
071public class LCG64ShiftRandom extends Random64 {
072

079        /**
080         * Parameter class for the {@code LCG64ShiftRandom} generator, for the
081         * parameters <i>a</i> and <i>b</i> of the LC recursion
082         * <i>r<sub>i+1</sub> = a · r<sub>i</sub> + b</i> mod <i>2<sup>64</sup></i>.
083         *
084         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
085         * @since 1.1
086         * @version 2.0
087         */
088        public static final class Param implements Serializable {
089
090                private static final long serialVersionUID = 1L;
091
092                /**
093                 * The default PRNG parameters: a = 0xFBD19FBBC5C07FF5L; b = 1
094                 */
095                public static final Param DEFAULT = Param.of(0xFBD19FBBC5C07FF5L, 1L);
096
097                /**
098                 * LEcuyer 1 parameters: a = 0x27BB2EE687B0B0FDL; b = 1
099                 */
100                public static final Param LECUYER1 = Param.of(0x27BB2EE687B0B0FDL, 1L);
101
102                /**
103                 * LEcuyer 2 parameters: a = 0x2C6FE96EE78B6955L; b = 1
104                 */
105                public static final Param LECUYER2 = Param.of(0x2C6FE96EE78B6955L, 1L);
106
107                /**
108                 * LEcuyer 3 parameters: a = 0x369DEA0F31A53F85L; b = 1
109                 */
110                public static final Param LECUYER3 = Param.of(0x369DEA0F31A53F85L, 1L);
111
112
113                /**
114                 * The parameter <i>a</i> of the LC recursion.
115                 */
116                public final long a;
117
118                /**
119                 * The parameter <i>b</i> of the LC recursion.
120                 */
121                public final long b;
122
123                /**
124                 * Create a new parameter object.
125                 *
126                 * @param a the parameter <i>a</i> of the LC recursion.
127                 * @param b the parameter <i>b</i> of the LC recursion.
128                 */
129                private Param(final long a, final long b) {
130                        this.a = a;
131                        this.b = b;
132                }
133
134                public static Param of(final long a, final long b) {
135                        return new Param(a, b);
136                }
137
138                @Override
139                public int hashCode() {
140                        return 31*(int)(a^(a >>> 32)) + 31*(int)(b^(b >>> 32));
141                }
142
143                @Override
144                public boolean equals(final Object obj) {
145                        return obj instanceof Param &&
146                                ((Param)obj).a == a &&
147                                ((Param)obj).b == b;
148                }
149
150                @Override
151                public String toString() {
152                        return format("%s[a=%d, b=%d]", getClass().getName(), a, b);
153                }

161        /**
162         * This class represents a <i>thread local</i> implementation of the
163         * {@code LCG64ShiftRandom} PRNG.
164         *
165         * It's recommended to initialize the {@code RandomRegistry} the following
166         * way:
167         *
168         * <pre>{@code
169         * // Register the PRNG with the default parameters.
170         * RandomRegistry.setRandom(new LCG64ShiftRandom.ThreadLocal());
171         *
172         * // Register the PRNG with the {@code LECUYER3} parameters.
173         * RandomRegistry.setRandom(new LCG64ShiftRandom.ThreadLocal(
174         *     LCG64ShiftRandom.LECUYER3
175         * ));
176         * }</pre>
177         *
178         * Be aware, that calls of the {@code setSeed(long)} method will throw an
179         * {@code UnsupportedOperationException} for <i>thread local</i> instances.
180         * <pre>{@code
181         * RandomRegistry.setRandom(new LCG64ShiftRandom.ThreadLocal());
182         *
183         * // Will throw 'UnsupportedOperationException'.
184         * RandomRegistry.getRandom().setSeed(1234);
185         * }</pre>
186         *
187         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
188         * @since 1.1
189         * @version 3.0
190         */
191        public static final class ThreadLocal
192                extends java.lang.ThreadLocal<LCG64ShiftRandom>
193        {
194                private static final long STEP_BASE = 1L << 56;
195
196                private int _block = 0;
197                private long _seed = PRNG.seed();
198
199                private final Param _param;
200
201                /**
202                 * Create a new <i>thread local</i> instance of the
203                 * {@code LCG64ShiftRandom} PRNG with the given parameters.
204                 *
205                 * @param param the LC parameters.
206                 * @throws NullPointerException if the given parameters are null.
207                 */
208                public ThreadLocal(final Param param) {
209                        _param = requireNonNull(param, "PRNG param must not be null.");
210                }
211
212                /**
213                 * Create a new <i>thread local</i> instance of the
214                 * {@code LCG64ShiftRandom} PRNG with the {@code DEFAULT} parameters.
215                 */
216                public ThreadLocal() {
217                        this(Param.DEFAULT);
218                }
219
220                /**
221                 * Create a new PRNG using <i>block splitting</i> for guaranteeing well
222                 * distributed PRN for every thread.
223                 *
224                 * <p>
225                 * <strong>Tina’s Random Number Generator Library</strong>
226                 * <br>
227                 * <em>Chapter 2. Pseudo-random numbers for parallel Monte Carlo
228                 *     simulations, Page 7</em>
229                 * <br>
230                 * <small>Heiko Bauke</small>
231                 * <br>
232                 * [<a href="http://numbercrunch.de/trng/trng.pdf">
233                 *  http://numbercrunch.de/trng/trng.pdf
234                 *  </a>].
235                 */
236                @Override
237                protected synchronized LCG64ShiftRandom initialValue() {
238                        if (_block > 127) {
239                                _block = 0;
240                                _seed = PRNG.seed();
241                        }
242
243                        final LCG64ShiftRandom random = new TLRandom(_param, _seed);
244                        random.jump(_block++*STEP_BASE);
245                        return random;
246                }
247
248        }

380         * Represents the state of this random engine
381         */
382        private final static class State implements Serializable {
383                private static final long serialVersionUID = 1L;
384
385                long _r;
386
387                State(final byte[] seed) {
388                        setSeed(seed);
389                }
390
391                void setSeed(final byte[] seed) {
392                        if (seed.length < SEED_BYTES) {
393                                throw new IllegalArgumentException(format(
394                                        "Required %d seed bytes, but got %d.",
395                                        SEED_BYTES, seed.length
396                                ));
397                        }
398
399                        _r = readLong(seed, 0);
400                }
401
402                @Override
403                public int hashCode() {
404                        return Long.hashCode(_r);
405                }
406
407                @Override
408                public boolean equals(final Object obj) {
409                        return obj instanceof State && ((State)obj)._r == _r;
410                }
411
412                @Override
413                public String toString() {
414                        return format("State[%d]", _r);
415                }

497        @Override
498        public long nextLong() {
499                step();
500
501                long t = _state._r;
502                t ^= t >>> 17;
503                t ^= t << 31;
504                t ^= t >>> 8;
505                return t;
506        }
507
508        private void step() {
509                _state._r = _param.a*_state._r + _param.b;
510        }

528        /**
529         * Changes the internal state of the PRNG in a way that future calls to
530         * {@link #nextLong()} will generated the s<sup>th</sup> sub-stream of
531         * p<sup>th</sup> sub-streams. <i>s</i> must be within the range of
532         * {@code [0, p-1)}. This method is mainly used for <i>parallelization</i>
533         * via <i>leap-frogging</i>.
534         *
535         * @param p the overall number of sub-streams
536         * @param s the s<sup>th</sup> sub-stream
537         * @throws IllegalArgumentException if {@code p < 1 || s >= p}.
538         */
539        public void split(final int p, final int s) {
540                if (p < 1) {
541                        throw new IllegalArgumentException(format(
542                                "p must be >= 1 but was %d.", p
543                        ));
544                }
545                if (s >= p) {
546                        throw new IllegalArgumentException(format(
547                                "s must be < %d but was %d.", p, s
548                        ));
549                }
550
551                if (p > 1) {
552                        jump(s + 1);
553                        final long b = _param.b*f(p, _param.a);
554                        final long a = IntMath.pow(_param.a, p);
555                        _param = Param.of(a, b);
556                        backward();
557                }
558        }
559
560        /**
561         * Changes the internal state of the PRNG in such a way that the engine
562         * <i>jumps</i> 2<sup>s</sup> steps ahead.
563         *
564         * @param s the 2<sup>s</sup> steps to jump ahead.
565         * @throws IllegalArgumentException if {@code s < 0}.
566         */
567        public void jump2(final int s) {
568                if (s < 0) {
569                        throw new IllegalArgumentException(format(
570                                "s must be positive but was %d.", s
571                        ));
572                }
573
574                if (s >= Long.SIZE) {
575                        throw new IllegalArgumentException(format(
576                                "The 'jump2' size must be smaller than %d but was %d.",
577                                Long.SIZE, s
578                        ));
579                }
580
581                _state._r = _state._r*IntMath.pow(_param.a, 1L << s) +
582                                        f(1L << s, _param.a)*_param.b;
583        }
584
585        /**
586         * Changes the internal state of the PRNG in such a way that the engine
587         * <i>jumps</i> s steps ahead.
588         *
589         * @param step the steps to jump ahead.
590         * @throws IllegalArgumentException if {@code s < 0}.
591         */
592        public void jump(final long step) {
593                if (step < 0) {
594                        throw new IllegalArgumentException(format(
595                                "step must be positive but was %d", step
596                        ));
597                }
598
599                if (step < 16) {
600                        for (int i = 0; i < step; ++i) {
601                                step();
602                        }
603                } else {
604                        long s = step;
605                        int i = 0;
606                        while (s > 0) {
607                                if (s%2 == 1) {
608                                        jump2(i);
609                                }
610                                ++i;
611                                s >>= 1;
612                        }
613                }
614        }
615
616        private void backward() {
617                for (int i = 0; i < Long.SIZE; ++i) {
618                        jump2(i);
619                }
620        }

663        /**
664         * Compute prod(1+a^(2^i), i=0..l-1).
665         */
666        private static long g(final int l, final long a) {
667                long p = a;
668                long res = 1;
669                for (int i = 0; i < l; ++i) {
670                        res *= 1 + p;
671                        p *= p;
672                }
673
674                return res;
675        }
676
677        /**
678         * Compute sum(a^i, i=0..s-1).
679         */
680        private static long f(final long s, final long a) {
681                long y = 0;
682
683                if (s != 0) {
684                        long e = log2Floor(s);
685                        long p = a;
686
687                        for (int l = 0; l <= e; ++l) {
688                                if (((1L << l) & s) != 0) {
689                                        y = g(l, a) + p*y;
690                                }
691                                p *= p;
692                        }
693                }
694
695                return y;
696        }
