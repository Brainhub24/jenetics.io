033/**
034 * This generator was discovered and characterized by George Marsaglia
035 * [<a href="http://www.jstatsoft.org/v08/i14/paper">Xorshift RNGs</a>]. In just
036 * three XORs and three shifts (generally fast operations) it produces a full
037 * period of 2<sup>64</sup> - 1 on 64 bits. (The missing value is zero, which
038 * perpetuates itself and must be avoided.) High and low bits pass Diehard.
039 * <p>
040 * Implementation of the XOR shift PRNG. The following listing shows the actual
041 * PRNG implementation.
042 * <pre>{@code
043 * private final int a, b, c = <param>
044 * private long x = <seed>
045 *
046 * long nextLong() {
047 *     x ^= x << a;
048 *     x ^= x >>> b;
049 *     return x ^= x << c;
050 * }
051 * }</pre>
052 *
053 * @see <a href="http://www.jstatsoft.org/v08/i14/paper">
054 *      Xorshift RNGs, George Marsaglia</a>
055 *
056 * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
057 * @since 1.0
058 * @version 1.0
059 */
060public class XOR64ShiftRandom extends Random64 {
061

453         * This class represents a <i>thread local</i> implementation of the
454         * {@code XOR64ShiftRandom} PRNG.

456         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
457         * @since 1.0
458         * @version 1.0

460        public static final class ThreadLocal
461                extends java.lang.ThreadLocal<XOR64ShiftRandom>
462        {
463                private final ParamSelector _selector;
464
465                /**
466                 * Create a new PRNG thread-local instance with the given <em>shift</em>
467                 * and parameter.
468                 *
469                 * @param shift the <em>shift</em> strategy of the PRNG
470                 * @param param the parameter of the PRNG.
471                 * @throws NullPointerException if the given {@code shift} or
472                 *        {@code param} is {@code null}.
473                 */
474                public ThreadLocal(final Shift shift, final Param param) {
475                        _selector = new ParamSelector(
476                                singletonList(shift),
477                                singletonList(param)
478                        );
479                }
480
481                /**
482                 * Create a new PRNG thread-local instance with the given <em>shift</em>
483                 * parameter.
484                 *
485                 * @param shift the <em>shift</em> strategy of the PRNG
486                 * @throws NullPointerException if the given {@code shift} is
487                 *         {@code null}.
488                 */
489                public ThreadLocal(final Shift shift) {
490                        _selector = new ParamSelector(singletonList(shift), Param.PARAMS);
491                }
492
493                /**
494                 * Create a new PRNG thread-local instance with the given parameter.
495                 *
496                 * @param param the parameter of the PRNG.
497                 * @throws NullPointerException if the given {@code param} is
498                 *         {@code null}.
499                 */
500                public ThreadLocal(final Param param) {
501                        _selector = new ParamSelector(
502                                asList(Shift.values()),
503                                singletonList(param)
504                        );
505                }
506
507                /**
508                 * Create a new PRNG thread-local instance.
509                 */
510                public ThreadLocal() {
511                        _selector = new ParamSelector(asList(Shift.values()), Param.PARAMS);
512                }
513
514                /**
515                 * Create a new PRNG using different parameter values for every thread.
516                 */
517                @Override
518                protected XOR64ShiftRandom initialValue() {
519                        final TLRandom random = new TLRandom(
520                                _selector.shift(),
521                                _selector.param(),
522                                _selector.seed()
523                        );
524                        _selector.next();
525
526                        return random;
527                }
528        }

530        /**
531         * Helper class for periodical change or the PRNG parameters <em>shift</em>,
532         * <em>param</em> and <em>seed</em>.
533         */
534        static final class ParamSelector {
535                private final List<Shift> _shifts;
536                private final List<Param> _params;
537                private final int _paramCount;
538
539                private int _index = 0;
540                private byte[] _seed = XOR32ShiftRandom.seedBytes();
541
542                ParamSelector(final List<Shift> shifts, final List<Param> params) {
543                        _shifts = requireNonNull(shifts);
544                        _params = requireNonNull(params);
545                        _paramCount = shifts.size()*params.size();
546                }
547
548                void next() {
549                        if (++_index >= _paramCount) {
550                                _index = 0;
551                                _seed = XOR32ShiftRandom.seedBytes();
552                        }
553                }
554
555                Shift shift() {
556                        return _shifts.get(_index/_params.size());
557                }
558
559                Param param() {
560                        return _params.get(_index%_params.size());
561                }
562
563                byte[] seed() {
564                        return _seed;
565                }
566        }
