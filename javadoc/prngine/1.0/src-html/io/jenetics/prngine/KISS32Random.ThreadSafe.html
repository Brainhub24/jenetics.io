029/**
030 * Implementation of an simple PRNG as proposed in
031 * <a href="http://www0.cs.ucl.ac.uk/staff/d.jones/GoodPracticeRNG.pdf">
032 * Good Practice in (Pseudo) Random Number Generation for Bioinformatics
033 * Applications</a> (JKISS32, page 3) by <em><a href="mailto:d.jones@cs.ucl.ac.uk">
034 * David Jones</a>, UCL Bioinformatics Group</em>.
035 * <p>
036 * The following listing shows the actual PRNG implementation.
037 * <pre>{@code
038 * private int x, y, z, w = <seed>
039 * private int c = 0;
040 *
041 * int nextInt() {
042 *     int t;
043 *     y ^= y << 5;
044 *     y ^= y >>> 7;
045 *     y ^= y << 22;
046 *
047 *     t = z + w + c;
048 *     z = w;
049 *     c = t >>> 31;
050 *     w = t & 2147483647;
051 *     x += 1411392427;
052 *
053 *     return x + y + w;
054 * }
055 * }</pre>
056 *
057 * <p>
058 * The period of this <i>PRNG</i> is &asymp; 2<sup>121</sup>
059 * &asymp; 2.6&sdot;10<sup>36</sup>
060 *
061 * <p>
062 * <strong>Not that the base implementation of the {@code KISS32Random}
063 * class is not thread-safe.</strong> If multiple threads requests random
064 * numbers from this class, it <i>must</i> be synchronized externally.
065 * Alternatively you can use the thread-safe implementations
066 * {@link KISS32Random.ThreadSafe} or {@link KISS32Random.ThreadLocal}.
067 *
068 * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
069 * @since 1.0
070 * @version 1.0
071 */
072public class KISS32Random extends Random32 {
073
074        private static final long serialVersionUID = 1L;
075
076        /**
077         * This class represents a <i>thread local</i> implementation of the
078         * {@code KISS32Random} PRNG.
079         *
080         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
081         * @since 1.0
082         * @version 1.0
083         */
084        public static final class ThreadLocal
085                extends java.lang.ThreadLocal<KISS32Random>
086        {
087
088                /**
089                 * Create a new PRNG using different seed values for every thread.
090                 */
091                @Override
092                protected KISS32Random initialValue() {
093                        return new TLRandom();
094                }
095        }
096
097        private static final class TLRandom extends KISS32Random {
098                private static final long serialVersionUID = 1L;
099
100                private final Boolean _sentry = Boolean.TRUE;
101
102                private TLRandom() {
103                        super(KISS32Random.seedBytes());
104                }
105
106                @Override
107                public void setSeed(final byte[] seed) {
108                        if (_sentry != null) {
109                                throw new UnsupportedOperationException(
110                                        "The 'setSeed' method is not supported " +
111                                                "for thread local instances."
112                                );
113                        }
114                }
115        }

118        /**
119         * This is a <i>thread safe</i> variation of the this PRNG&mdash;by
120         * synchronizing the random number generation.
121         *
122         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
123         * @since 1.0
124         * @version 1.0
125         */
126        public static final class ThreadSafe extends KISS32Random {
127                private static final long serialVersionUID = 1L;
128
129                /**
130                 * Create a new thread-safe instance of the {@code KISS32Random}
131                 * engine.
132                 *
133                 * <pre>{@code
134                 * final byte[] seed = KISS32Random.seedBytes();
135                 * final Random random = new KISS32Random.ThreadSafe(seed);
136                 * }</pre>
137                 *
138                 * @see #seedBytes()
139                 *
140                 * @param seed the random seed value. The seed must be (at least)
141                 *        {@link #SEED_BYTES} long.
142                 * @throws IllegalArgumentException if the given seed is shorter than
143                 *         {@link #SEED_BYTES}
144                 */
145                public ThreadSafe(final byte[] seed) {
146                        super(seed);
147                }
148
149                /**
150                 * Create a new thread-safe instance of the {@code KISS32Random}
151                 * engine. The constructed PRNG is equivalent with
152                 * <pre>{@code
153                 * final long seed = ...;
154                 * final Random random = new KISS32Random.ThreadSafe();
155                 * random.setSeed(seed);
156                 * }</pre>
157                 * which is there for compatibility reasons with the Java {@link Random}
158                 * engine.
159                 *
160                 * @param seed the random seed value
161                 */
162                public ThreadSafe(final long seed) {
163                        super(seed);
164                }
165
166                /**
167                 * Create a new thread-safe instance of the {@code KISS32Random}
168                 * engine. The PRNG is initialized with {@link #seedBytes()}.
169                 */
170                public ThreadSafe() {
171                }
172
173                @Override
174                public synchronized void setSeed(final byte[] seed) {
175                        super.setSeed(seed);
176                }
177
178                @Override
179                public synchronized void setSeed(final long seed) {
180                        super.setSeed(seed);
181                }
182
183                @Override
184                public synchronized int nextInt() {
185                        return super.nextInt();
186                }
187
188        }
189
190
191        /**
192         * The internal state of random engine.
193         *
194         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
195         * @since 1.0
196         * @version 1.0
197         */
198        private static final class State implements Serializable {
199                private static final long serialVersionUID = 1L;
200
201                int _x = 123456789;
202                int _y = 234567891;
203                int _z = 345678912;
204                int _w = 456789123;
205                int _c = 0;
206
207                State(final byte[] seed) {
208                        setSeed(seed);
209                }
210
211                void setSeed(final byte[] seed) {
212                        if (seed.length < SEED_BYTES) {
213                                throw new IllegalArgumentException(format(
214                                        "Required %d seed bytes, but got %d.",
215                                        SEED_BYTES, seed.length
216                                ));
217                        }
218
219                        _x = readInt(seed, 0);
220                        _y = readInt(seed, 1);
221                        _z = readInt(seed, 2);
222                        _w = readInt(seed, 3);
223                }
224
225                @Override
226                public int hashCode() {
227                        int hash = 31;
228                        hash += 37*_x + 17;
229                        hash += 37*_y + 17;
230                        hash += 37*_z + 17;
231                        hash += 37*_w + 17;
232                        hash += 37*_c + 17;
233
234                        return hash;
235                }
236
237                @Override
238                public boolean equals(final Object obj) {
239                        return obj instanceof State &&
240                                _x == ((State)obj)._x &&
241                                _y == ((State)obj)._y &&
242                                _z == ((State)obj)._z &&
243                                _w == ((State)obj)._w &&
244                                _c == ((State)obj)._c;
245                }
246
247                @Override
248                public String toString() {
249                        return format("State[%d, %d, %d, %d, %d]", _x, _y, _z, _w, _c);
250                }
251
252        }
253
254        /* *************************************************************************
255         * Main class.
256         * ************************************************************************/
257
258        /**
259         * The number of seed bytes (16) this PRNG requires.
260         */
261        public static final int SEED_BYTES = 16;
262
263        private final State _state;
264
265        /**
266         * Create a new <em>not</em> thread-safe instance of the {@code KISS32Random}
267         * engine.
268         *
269         * <pre>{@code
270         * final byte[] seed = KISS32Random.seedBytes();
271         * final Random random = new KISS32Random(seed);
272         * }</pre>
273         *
274         * @see #seedBytes()
275         *
276         * @param seed the random seed value. The seed must be (at least)
277         *        {@link #SEED_BYTES} long.
278         * @throws IllegalArgumentException if the given seed is shorter than
279         *         {@link #SEED_BYTES}
280         */
281        public KISS32Random(final byte[] seed) {
282                _state = new State(seed);
283        }
284
285        /**
286         * Create a new <em>not</em> thread-safe instance of the {@code KISS32Random}
287         * engine. The constructed PRNG is equivalent with
288         * <pre>{@code
289         * final long seed = ...;
290         * final Random random = new KISS32Random();
291         * random.setSeed(seed);
292         * }</pre>
293         * which is there for compatibility reasons with the Java {@link Random}
294         * engine.
295         *
296         * @param seed the random seed value
297         */
298        public KISS32Random(final long seed) {
299                this(seedBytes(seed, SEED_BYTES));
300        }
301
302        /**
303         * Create a new <em>not</em> thread-safe instance of the {@code KISS32Random}
304         * engine. The PRNG is initialized with {@link #seedBytes()}.
305         */
306        public KISS32Random() {
307                this(seedBytes());
308        }
309
310        @Override
311        public int nextInt() {
312                step();
313                return _state._x + _state._y + _state._w;
314        }
315
316        private void step() {
317                _state._y ^= _state._y << 5;
318                _state._y ^= _state._y >>> 7;
319                _state._y ^= _state._y << 22;
320
321                final int t = _state._z + _state._w + _state._c;
322                _state._z = _state._w;
323                _state._c = t >>> 31;
324                _state._w = t&2147483647;
325                _state._x += 1411392427;
326        }
327
328        /**
329         * Initializes the PRNg with the given seed.
330         *
331         * @see #seedBytes()
332         *
333         * @param seed the random seed value. The seed must be (at least)
334         *        {@link #SEED_BYTES} big.
335         * @throws IllegalArgumentException if the given seed is shorter than
336         *         {@link #SEED_BYTES}
337         */
338        public void setSeed(final byte[] seed) {
339                if (_state != null) _state.setSeed(seed);
340        }
341
342        @Override
343        public void setSeed(final long seed) {
344                setSeed(PRNG.seedBytes(seed, SEED_BYTES));
345        }
346
347        @Override
348        public int hashCode() {
349                int hash = 37;
350                hash += 31*_state.hashCode() + 17;
351                return hash;
352        }
353
354        @Override
355        public boolean equals(final Object obj) {
356                return obj instanceof KISS32Random &&
357                        Objects.equals(((KISS32Random)obj)._state, _state);
358        }
359
360        @Override
361        public String toString() {
362                return format("%s[%s]", getClass().getSimpleName(), _state);
363        }
364
365        /**
366         * Create a new <em>seed</em> byte array suitable for this PRNG. The
367         * returned seed array is {@link #SEED_BYTES} long.
368         *
369         * @see PRNG#seedBytes(int)
370         *
371         * @return a new <em>seed</em> byte array of length {@link #SEED_BYTES}
372         */
373        public static byte[] seedBytes() {
374                return PRNG.seedBytes(SEED_BYTES);
375        }
376
377}
