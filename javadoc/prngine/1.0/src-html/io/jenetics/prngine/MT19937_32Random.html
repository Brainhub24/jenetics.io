030/**
031 * This is a 32-bit version of Mersenne Twister pseudorandom number generator.
032 * <p>
033 * <i>
034 * References: M. Matsumoto and T. Nishimura, "Mersenne Twister: a
035 * 623-dimensionally equidistributed uniform pseudorandom number generator".
036 * ACM Transactions on Modeling and Computer Simulation 8. (Jan. 1998) 3--30.
037 * </i>
038 * <p>
039 * <em>
040 * This is a port of the
041 * <a href="https://github.com/rabauke/trng4/blob/master/src/mt19937.hpp">
042 * trng::mt19937</a> PRNG class of the
043 * <a href="http://numbercrunch.de/trng/">TRNG</a> library created by Heiko
044 * Bauke.</em>
045 *
046 * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
047 * @since 1.0
048 * @version 1.0
049 */
050public class MT19937_32Random extends Random32 {
051
052        private static final long serialVersionUID = 1L;
053
054        private static final int N = 624;
055        private static final int M = 397;
056
057        private static final int UM = 0x80000000; // most significant bit
058        private static final int LM = 0x7FFFFFFF; // least significant 31 bits
059
060        private static final int[] MAG01 = {0, 0x9908b0df};
061
062        /**
063         * This class represents a <i>thread local</i> implementation of the
064         * {@code MT19937_32Random} PRNG.
065         *
066         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
067         * @since 1.0
068         * @version 1.0
069         */
070        public static class ThreadLocal
071                extends java.lang.ThreadLocal<MT19937_32Random>
072        {
073
074                /**
075                 * Create a new PRNG using different seed values for every thread.
076                 */
077                @Override
078                protected MT19937_32Random initialValue() {
079                        return new TLRandom(PRNG.seed());
080                }
081        }
082
083        private static final class TLRandom extends MT19937_32Random {
084                private static final long serialVersionUID = 1L;
085
086                private final Boolean _sentry = Boolean.TRUE;
087
088                private TLRandom(final long seed) {
089                        super(seed);
090                }
091
092                @Override
093                public void setSeed(final long seed) {
094                        if (_sentry != null) {
095                                throw new UnsupportedOperationException(
096                                        "The 'setSeed(long)' method is not supported " +
097                                                "for thread local instances."
098                                );
099                        }
100                }
101
102        }

104        /**
105         * This is a <i>thread safe</i> variation of the this PRNG&mdash;by
106         * synchronizing the random number generation.
107         *
108         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
109         * @since 1.0
110         * @version 1.0
111         */
112        public static final class ThreadSafe extends MT19937_32Random {
113                private static final long serialVersionUID = 1L;
114
115                /**
116                 * Create a new thread-safe instance of the {@code MT19937_32Random}
117                 * engine.
118                 *
119                 * <pre>{@code
120                 * final byte[] seed = MT19937_32Random.seedBytes();
121                 * final Random random = new MT19937_32Random.ThreadSafe(seed);
122                 * }</pre>
123                 *
124                 * @see #seedBytes()
125                 *
126                 * @param seed the random seed value. The seed must be (at least)
127                 *        {@link #SEED_BYTES} long.
128                 * @throws IllegalArgumentException if the given seed is shorter than
129                 *         {@link #SEED_BYTES}
130                 */
131                public ThreadSafe(final byte[] seed) {
132                        super(seed);
133                }
134
135                /**
136                 * Create a new thread-safe instance of the {@code MT19937_32Random}
137                 * engine. The constructed PRNG is equivalent with
138                 * <pre>{@code
139                 * final long seed = ...;
140                 * final Random random = new MT19937_32Random.ThreadSafe();
141                 * random.setSeed(seed);
142                 * }</pre>
143                 * which is there for compatibility reasons with the Java {@link Random}
144                 * engine.
145                 *
146                 * @param seed the random seed value
147                 */
148                public ThreadSafe(final long seed) {
149                        super(seed);
150                }
151
152                /**
153                 * Create a new thread-safe instance of the {@code MT19937_32Random}
154                 * engine. The PRNG is initialized with {@link #seedBytes()}.
155                 */
156                public ThreadSafe() {
157                }
158
159                @Override
160                public synchronized void setSeed(final byte[] seed) {
161                        super.setSeed(seed);
162                }
163
164                @Override
165                public synchronized void setSeed(final long seed) {
166                        super.setSeed(seed);
167                }
168
169                @Override
170                public synchronized int nextInt() {
171                        return super.nextInt();
172                }
173
174        }

177        /**
178         * The internal state of this PRNG.
179         */
180        private static final class State implements Serializable {
181                private static final long serialVersionUID = 1L;
182
183                int mti = 0;
184                int[] mt = new int[N];
185
186                State(final byte[] seed) {
187                        setSeed(seed);
188                }
189
190                State(final long seed) {
191                        setSeed(seed);
192                }
193
194                void setSeed(final byte[] seed) {
195                        if (seed.length < SEED_BYTES) {
196                                throw new IllegalArgumentException(format(
197                                        "Required %d seed bytes, but got %d.",
198                                        SEED_BYTES, seed.length
199                                ));
200                        }
201
202                        for (mti = 0; mti < N; ++mti) {
203                                mt[mti] = readInt(seed, mti);
204                        }
205                }
206
207                void setSeed(final long seed) {
208                        mt[0] = (int)seed;
209                        for (mti = 1; mti < N; ++mti) {
210                                mt[mti] = 1812433253*(mt[mti - 1]^(mt[mti - 1] >>> 30)) + mti;
211                        }
212                }
213
214                @Override
215                public int hashCode() {
216                        int hash = 17;
217                        hash += 37*mti + 31;
218                        hash += 37*Arrays.hashCode(mt) + 31;
219                        return hash;
220                }
221
222                @Override
223                public boolean equals(final Object obj) {
224                        return obj instanceof State &&
225                                mti == ((State)obj).mti &&
226                                Arrays.equals(mt, ((State)obj).mt);
227                }
228        }

231        /* *************************************************************************
232         * Main class.
233         * ************************************************************************/
234
235        /**
236         * The number of seed bytes (2,496) this PRNG requires.
237         */
238        public static final int SEED_BYTES = N*Integer.BYTES;
239
240        private final State _state;
241
242        /**
243         * Create a new <em>not</em> thread-safe instance of the
244         * {@code MT19937_32Random} engine.
245         *
246         * <pre>{@code
247         * final byte[] seed = MT19937_32Random.seedBytes();
248         * final Random random = new MT19937_32Random(seed);
249         * }</pre>
250         *
251         * @see #seedBytes()
252         *
253         * @param seed the random seed value. The seed must be (at least)
254         *        {@link #SEED_BYTES} long.
255         * @throws IllegalArgumentException if the given seed is shorter than
256         *         {@link #SEED_BYTES}
257         */
258        public MT19937_32Random(final byte[] seed) {
259                _state = new State(seed);
260        }
261
262        /**
263         * Create a new <em>not</em> thread-safe instance of the
264         * {@code MT19937_32Random} engine. The constructed PRNG is equivalent with
265         * <pre>{@code
266         * final long seed = ...;
267         * final Random random = new MT19937_32Random();
268         * random.setSeed(seed);
269         * }</pre>
270         * which is there for compatibility reasons with the Java {@link Random}
271         * engine.
272         *
273         * @param seed the random seed value
274         */
275        public MT19937_32Random(final long seed) {
276                _state = new State(seed);
277        }
278
279        /**
280         * Return a new random engine with a safe seed value.
281         */
282        public MT19937_32Random() {
283                this(seedBytes());
284        }
285
286        @Override
287        public int nextInt() {
288                int x;
289
290                // Generate N words at one time.
291                if (_state.mti >= N) {
292                        int i = 0;
293                        for (i = 0; i < N - M; ++i) {
294                                x = (_state.mt[i] & UM) | (_state.mt[i + 1] & LM);
295                                _state.mt[i] = _state.mt[i + M]^(x >>> 1)^MAG01[x & 1];
296                        }
297                        for (; i < N - 1; ++i) {
298                                x = (_state.mt[i] & UM) | (_state.mt[i + 1] & LM);
299                                _state.mt[i] = _state.mt[i + (M - N)]^(x >>> 1)^MAG01[x & 1];
300                        }
301
302                        x = (_state.mt[N - 1] & UM)|(_state.mt[0] & LM);
303                        _state.mt[N - 1] = _state.mt[M - 1]^(x >>> 1)^MAG01[x & 1];
304                        _state.mti = 0;
305                }
306
307                x = _state.mt[_state.mti++];
308                x ^= x >>> 11;
309                x ^= (x << 7) & 0x9d2c5680;
310                x ^= (x << 15) & 0xefc60000;
311                x ^= x >>> 18;
312
313                return x;
314        }
315
316        /**
317         * Initializes the PRNg with the given seed.
318         *
319         * @see #seedBytes()
320         *
321         * @param seed the random seed value. The seed must be (at least)
322         *        {@link #SEED_BYTES} big.
323         * @throws IllegalArgumentException if the given seed is shorter than
324         *         {@link #SEED_BYTES}
325         */
326        public void setSeed(final byte[] seed) {
327                if (_state != null) _state.setSeed(seed);
328        }
329
330        @Override
331        public void setSeed(final long seed) {
332                if (_state != null) _state.setSeed(seed);
333        }
334
335        @Override
336        public int hashCode() {
337                int hash = 37;
338                hash += 31*_state.hashCode() + 17;
339                return hash;
340        }
341
342        @Override
343        public boolean equals(final Object obj) {
344                return obj instanceof MT19937_32Random &&
345                        Objects.equals(((MT19937_32Random)obj)._state, _state);
346        }
347
348        /**
349         * Create a new <em>seed</em> byte array suitable for this PRNG. The
350         * returned seed array is {@link #SEED_BYTES} long.
351         *
352         * @see PRNG#seedBytes(int)
353         *
354         * @return a new <em>seed</em> byte array of length {@link #SEED_BYTES}
355         */
356        public static byte[] seedBytes() {
357                return PRNG.seedBytes(SEED_BYTES);
358        }
