030/**
031 * Implementation of an simple PRNG as proposed in
032 * <a href="http://www0.cs.ucl.ac.uk/staff/d.jones/GoodPracticeRNG.pdf">
033 * Good Practice in (Pseudo) Random Number Generation for Bioinformatics
034 * Applications</a> (JKISS64, page 10) by <em><a href="mailto:d.jones@cs.ucl.ac.uk">
035 * David Jones</a>, UCL Bioinformatics Group</em>.
036 *
037 * <p>
038 * The following listing shows the actual PRNG implementation.
039 * <pre>{@code
040 * private long x, y = <seed>
041 * private int z1, c1, z2, c2 = <seed>
042 *
043 * long nextLong() {
044 *     x = 0x14ADA13ED78492ADL*x + 123456789;
045 *
046 *     y ^= y << 21;
047 *     y ^= y >>> 17;
048 *     y ^= y << 30;
049 *
050 *     long t = 4294584393L*_z1 + _c1;
051 *     c1 = (int)(t >> 32);
052 *     z1 = (int)t;
053 *
054 *     t = 4246477509L*z2 + c2;
055 *     c2 = (int)(t >> 32);
056 *     z2 = (int)t;
057 *
058 *     return x + y + z1 + ((long)z2 << 32);
059 * }
060 * }</pre>
061 *
062 * <p>
063 * The period of this <i>PRNG</i> is &asymp; 2<sup>250</sup>
064 * &asymp; 1.8&sdot;10<sup>75</sup>
065 *
066 * <p>
067 * <strong>Not that the base implementation of the {@code KISS64Random}
068 * class is not thread-safe.</strong> If multiple threads requests random
069 * numbers from this class, it <i>must</i> be synchronized externally.
070 * Alternatively you can use the thread-safe implementations
071 * {@link KISS64Random.ThreadSafe} or {@link KISS64Random.ThreadLocal}.
072 *
073 * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
074 * @since 1.0
075 * @version 1.0
076 */
077public class KISS64Random extends Random64 {
078
079        private static final long serialVersionUID = 1L;
080
081        /**
082         * This class represents a <i>thread local</i> implementation of the
083         * {@code KISS64Random} PRNG.
084         *
085         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
086         * @since 1.0
087         * @version 1.0
088         */
089        public static final class ThreadLocal
090                extends java.lang.ThreadLocal<KISS64Random>
091        {
092
093                /**
094                 * Create a new PRNG using different seed values for every thread.
095                 */
096                @Override
097                protected KISS64Random initialValue() {
098                        return new TLRandom(seedBytes());
099                }
100        }

102        private static final class TLRandom extends KISS64Random {
103                private static final long serialVersionUID = 1L;
104
105                private final Boolean _sentry = Boolean.TRUE;
106
107                private TLRandom(final byte[] seed) {
108                        super(seed);
109                }
110
111                @Override
112                public void setSeed(final byte[] seed) {
113                        if (_sentry != null) {
114                                throw new UnsupportedOperationException(
115                                        "The 'setSeed' method is not supported " +
116                                                "for thread local instances."
117                                );
118                        }
119                }
120        }

123        /**
124         * This is a <i>thread safe</i> variation of the this PRNG&mdash;by
125         * synchronizing the random number generation.
126         *
127         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
128         * @since 1.0
129         * @version 1.0
130         */
131        public static final class ThreadSafe extends KISS64Random {
132                private static final long serialVersionUID = 1L;
133
134                /**
135                 * Create a new thread-safe instance of the {@code KISS64Random} engine.
136                 *
137                 * <pre>{@code
138                 * final byte[] seed = KISS64Random.seedBytes();
139                 * final Random random = new KISS64Random.ThreadSafe(seed);
140                 * }</pre>
141                 *
142                 * @see #seedBytes()
143                 *
144                 * @param seed the random seed value. The seed must be (at least)
145                 *        {@link #SEED_BYTES} long.
146                 * @throws IllegalArgumentException if the given seed is shorter than
147                 *         {@link #SEED_BYTES}
148                 */
149                public ThreadSafe(final byte[] seed) {
150                        super(seed);
151                }
152
153                /**
154                 * Create a new thread-safe instance of the {@code KISS64Random} engine.
155                 * The constructed PRNG is equivalent with
156                 * <pre>{@code
157                 * final long seed = ...;
158                 * final Random random = new KISS64Random.ThreadSafe();
159                 * random.setSeed(seed);
160                 * }</pre>
161                 * which is there for compatibility reasons with the Java {@link Random}
162                 * engine.
163                 *
164                 * @param seed the random seed value
165                 */
166                public ThreadSafe(final long seed) {
167                        super(seed);
168                }
169
170                /**
171                 * Create a new thread-safe instance of the {@code KISS64Random} engine.
172                 * The PRNG is initialized with {@link #seedBytes()}.
173                 */
174                public ThreadSafe() {
175                }
176
177                @Override
178                public synchronized void setSeed(final byte[] seed) {
179                        super.setSeed(seed);
180                }
181
182                @Override
183                public synchronized void setSeed(final long seed) {
184                        super.setSeed(seed);
185                }
186
187                @Override
188                public synchronized long nextLong() {
189                        return super.nextLong();
190                }
191        }
192
193        /**
194         * The internal state of this PRNG.
195         */
196        private static final class State implements Serializable {
197                private static final long serialVersionUID = 1L;
198
199                long _x = 123456789123L;
200                long _y = 987654321987L;
201                int _z1 = 43219876;
202                int _c1 = 6543217;
203                int _z2 = 21987643;
204                int _c2 = 1732654;
205
206                State(final byte[] seed) {
207                        setSeed(seed);
208                }
209
210                void setSeed(final byte[] seed) {
211                        if (seed.length < SEED_BYTES) {
212                                throw new IllegalArgumentException(format(
213                                        "Required %d seed bytes, but got %d.",
214                                        SEED_BYTES, seed.length
215                                ));
216                        }
217
218                        _x = readLong(seed, 0);
219                        _y = readLong(seed, 1);
220                        _z1 = readInt(seed, 4);
221                        _c1 = readInt(seed, 5);
222                        _z2 = readInt(seed, 6);
223                        _c2 = readInt(seed, 7);
224                }
225
226                @Override
227                public int hashCode() {
228                        int hash = 31;
229                        hash += 37*_x + 17;
230                        hash += 37*_y + 17;
231                        hash += 37*_z1 + 17;
232                        hash += 37*_c1 + 17;
233                        hash += 37*_z2 + 17;
234                        hash += 37*_c2 + 17;
235
236                        return hash;
237                }
238
239                @Override
240                public boolean equals(final Object obj) {
241                        return obj instanceof State &&
242                                _x == ((State)obj)._x &&
243                                _y == ((State)obj)._y &&
244                                _z1 == ((State)obj)._z1 &&
245                                _c1 == ((State)obj)._c1 &&
246                                _z2 == ((State)obj)._z2 &&
247                                _c2 == ((State)obj)._c2;
248                }
249
250                @Override
251                public String toString() {
252                        return format(
253                                "State[%d, %d, %d, %d, %d, %d]",
254                                _x, _y, _z1, _c1, _z2, _c2
255                        );
256                }
257        }

260        /* *************************************************************************
261         * Main class.
262         * ************************************************************************/
263
264        /**
265         * The number of seed bytes (32) this PRNG requires.
266         */
267        public static final int SEED_BYTES = 32;
268
269        private final State _state;
270
271        /**
272         * Create a new <em>not</em> thread-safe instance of the {@code KISS64Random}
273         * engine.
274         *
275         * <pre>{@code
276         * final byte[] seed = KISS64Random.seedBytes();
277         * final Random random = new KISS64Random(seed);
278         * }</pre>
279         *
280         * @see #seedBytes()
281         *
282         * @param seed the random seed value. The seed must be (at least)
283         *        {@link #SEED_BYTES} long.
284         * @throws IllegalArgumentException if the given seed is shorter than
285         *         {@link #SEED_BYTES}
286         */
287        public KISS64Random(final byte[] seed) {
288                _state = new State(seed);
289        }
290
291        /**
292         * Create a new <em>not</em> thread-safe instance of the {@code KISS64Random}
293         * engine. The constructed PRNG is equivalent with
294         * <pre>{@code
295         * final long seed = ...;
296         * final Random random = new KISS64Random();
297         * random.setSeed(seed);
298         * }</pre>
299         * which is there for compatibility reasons with the Java {@link Random}
300         * engine.
301         *
302         * @param seed the random seed value
303         */
304        public KISS64Random(final long seed) {
305                this(PRNG.seedBytes(seed, SEED_BYTES));
306        }
307
308        /**
309         * Create a new <em>not</em> thread-safe instance of the {@code KISS64Random}
310         * engine. The PRNG is initialized with {@link #seedBytes()}.
311         */
312        public KISS64Random() {
313                this(seedBytes());
314        }
315
316        @Override
317        public long nextLong() {
318                step();
319                return _state._x + _state._y + _state._z1 + ((long)_state._z2 << 32);
320        }
321
322        private void step() {
323                _state._x = 0x14ADA13ED78492ADL*_state._x + 123456789;
324
325                _state._y ^= _state._y << 21;
326                _state._y ^= _state._y >>> 17;
327                _state._y ^= _state._y << 30;
328
329                long t = 4294584393L*_state._z1 + _state._c1;
330                _state._c1 = (int)(t >> 32);
331                _state._z1 = (int)t;
332
333                t = 4246477509L*_state._z2 + _state._c2;
334                _state._c2 = (int)(t >> 32);
335                _state._z2 = (int)t;
336        }
337
338        /**
339         * Initializes the PRNg with the given seed.
340         *
341         * @see #seedBytes()
342         *
343         * @param seed the random seed value. The seed must be (at least)
344         *        {@link #SEED_BYTES} big.
345         * @throws IllegalArgumentException if the given seed is shorter than
346         *         {@link #SEED_BYTES}
347         */
348        public void setSeed(final byte[] seed) {
349                if (_state != null) _state.setSeed(seed);
350        }
351
352        @Override
353        public void setSeed(final long seed) {
354                setSeed(PRNG.seedBytes(seed, SEED_BYTES));
355        }
356
357        @Override
358        public int hashCode() {
359                int hash = 37;
360                hash += 31*_state.hashCode() + 17;
361                return hash;
362        }
363
364        @Override
365        public boolean equals(final Object obj) {
366                return obj instanceof KISS64Random &&
367                        Objects.equals(((KISS64Random)obj)._state, _state);
368        }
369
370        @Override
371        public String toString() {
372                return format("%s[%s]", getClass().getSimpleName(), _state);
373        }
374
375        /**
376         * Create a new <em>seed</em> byte array suitable for this PRNG. The
377         * returned seed array is {@link #SEED_BYTES} long.
378         *
379         * @see PRNG#seedBytes(int)
380         *
381         * @return a new <em>seed</em> byte array of length {@link #SEED_BYTES}
382         */
383        public static byte[] seedBytes() {
384                return PRNG.seedBytes(SEED_BYTES);
385        }
