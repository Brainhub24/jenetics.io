033/**
034 * This generator was discovered and characterized by George Marsaglia
035 * [<a href="http://www.jstatsoft.org/v08/i14/paper">Xorshift RNGs</a>]. In just
036 * three XORs and three shifts (generally fast operations) it produces a full
037 * period of 2<sup>32</sup> - 1 on 32 bits. (The missing value is zero, which
038 * perpetuates itself and must be avoided.) High and low bits pass Diehard.
039 * <p>
040 * Implementation of the XOR shift PRNG. The following listing shows the actual
041 * PRNG implementation.
042 * <pre>{@code
043 * private final int a, b, c = <param>
044 * private int x = <seed>
045 *
046 * int nextInt() {
047 *     x ^= x << a;
048 *     x ^= x >>> b;
049 *     return x ^= x << c;
050 * }
051 * }</pre>
052 *
053 * @see <a href="http://www.jstatsoft.org/v08/i14/paper">
054 *      Xorshift RNGs, George Marsaglia</a>
055 *
056 * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
057 * @since 1.0
058 * @version 1.0
059 */
060public class XOR32ShiftRandom extends Random32 {
061

068        /**
069         * Enumeration of the different <em>shift</em> strategies.
070         *
071         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
072         * @since 1.0
073         * @version 1.0
074         */
075        public static enum Shift {
076
077                /**
078                 * Shift strategy number one.
079                 * <pre>{@code
080                 * int shift(int x, final Param param) {
081                 *     x ^= x << param.a;
082                 *     x ^= x >>> param.b;
083                 *     return x^x << param.c;
084                 * }
085                 * }</pre>
086                 */
087                SHIFT_1 {
088                        @Override
089                        public int shift(int x, final Param param) {
090                                x ^= x << param.a;
091                                x ^= x >>> param.b;
092                                return x^x << param.c;
093                        }
094                },
095
096                /**
097                 * Shift strategy number two.
098                 * <pre>{@code
099                 * int shift(int x, final Param param) {
100                 *     x ^= x << param.c;
101                 *     x ^= x >>> param.b;
102                 *     return x^x << param.a;
103                 * }
104                 * }</pre>
105                 */
106                SHIFT_2 {
107                        @Override
108                        public int shift(int x, final Param param) {
109                                x ^= x << param.c;
110                                x ^= x >>> param.b;
111                                return x^x << param.a;
112                        }
113                },
114
115                /**
116                 * Shift strategy number three.
117                 * <pre>{@code
118                 * int shift(int x, final Param param) {
119                 *     x ^= x >>> param.a;
120                 *     x ^= x << param.b;
121                 *     return x^x >>> param.c;
122                 * }
123                 * }</pre>
124                 */
125                SHIFT_3 {
126                        @Override
127                        public int shift(int x, final Param param) {
128                                x ^= x >>> param.a;
129                                x ^= x << param.b;
130                                return x^x >>> param.c;
131                        }
132                },
133
134                /**
135                 * Shift strategy number four.
136                 * <pre>{@code
137                 * int shift(int x, final Param param) {
138                 *     x ^= x >>> param.c;
139                 *     x ^= x << param.b;
140                 *     return x^x >>> param.a;
141                 * }
142                 * }</pre>
143                 */
144                SHIFT_4 {
145                        @Override
146                        public int shift(int x, final Param param) {
147                                x ^= x >>> param.c;
148                                x ^= x << param.b;
149                                return x^x >>> param.a;
150                        }
151                },
152
153                /**
154                 * Shift strategy number five.
155                 * <pre>{@code
156                 * int shift(int x, final Param param) {
157                 *     x ^= x << param.a;
158                 *     x ^= x << param.c;
159                 *     return x^x >>> param.b;
160                 * }
161                 * }</pre>
162                 */
163                SHIFT_5 {
164                        @Override
165                        public int shift(int x, final Param param) {
166                                x ^= x << param.a;
167                                x ^= x << param.c;
168                                return x^x >>> param.b;
169                        }
170                },
171
172                /**
173                 * Shift strategy number six.
174                 * <pre>{@code
175                 * int shift(int x, final Param param) {
176                 *     x ^= x << param.c;
177                 *     x ^= x << param.a;
178                 *     return x^x >>> param.b;
179                 * }
180                 * }</pre>
181                 */
182                SHIFT_6 {
183                        @Override
184                        public int shift(int x, final Param param) {
185                                x ^= x << param.c;
186                                x ^= x << param.a;
187                                return x^x >>> param.b;
188                        }
189                },
190
191                /**
192                 * Shift strategy number seven.
193                 * <pre>{@code
194                 * int shift(int x, final Param param) {
195                 *     x ^= x >>> param.a;
196                 *     x ^= x >>> param.c;
197                 *     return x^x << param.b;
198                 * }
199                 * }</pre>
200                 */
201                SHIFT_7 {
202                        @Override
203                        public int shift(int x, final Param param) {
204                                x ^= x >>> param.a;
205                                x ^= x >>> param.c;
206                                return x^x << param.b;
207                        }
208                },
209
210                /**
211                 * Shift strategy number eight.
212                 * <pre>{@code
213                 * int shift(int x, final Param param) {
214                 *     x ^= x >>> param.c;
215                 *     x ^= x >>> param.a;
216                 *     return x^x << param.b;
217                 * }
218                 * }</pre>
219                 */
220                SHIFT_8 {
221                        @Override
222                        public int shift(int x, final Param param) {
223                                x ^= x >>> param.c;
224                                x ^= x >>> param.a;
225                                return x^x << param.b;
226                        }
227                };
228
229                /**
230                 * The <em>default</em> shift strategy.
231                 */
232                public static final Shift DEFAULT = SHIFT_1;
233
234                /**
235                 * Performs the <em>xor</em> shift of {@code x} with the given
236                 * {@code param}.
237                 *
238                 * @param x the value where the <em>xor</em> shift is performed
239                 * @param param the shift parameters
240                 * @return the <em>xor</em> shifted value
241                 */
242                public abstract int shift(int x, final Param param);
243        }

271                 * The default parameter used by the PRNG. It's the parameter with the
272                 * best <i>dieharder</i> test result.

276                /**
277                 * The parameter <em>a</em>.
278                 */
279                public final int a;
280
281                /**
282                 * The parameter <em>b</em>.
283                 */
284                public final int b;
285
286                /**
287                 * The parameter <em>c</em>.
288                 */
289                public final int c;
290
291                /**
292                 * Parameter object for the parameters used by this PRNG.
293                 *
294                 * @param a first shift parameter
295                 * @param b second shift parameter
296                 * @param c third shift parameter
297                 */
298                public Param(final int a, final int b, final int c) {
299                        this.a = a;
300                        this.b = b;
301                        this.c = c;
302                }
303
304                @Override
305                public int hashCode() {
306                        int hash = 17;
307                        hash += 31*a + 37;
308                        hash += 31*b + 37;
309                        hash += 31*c + 37;
310
311                        return hash;
312                }
313
314                @Override
315                public boolean equals(final Object obj) {
316                        return  obj instanceof Param &&
317                                ((Param)obj).a == a &&
318                                ((Param)obj).b == b &&
319                                ((Param)obj).c == c;
320                }
321
322                @Override
323                public String toString() {
324                        return String.format("Param(%d, %d, %d)", a, b, c);
325                }
326        }

332        /**
333         * This class represents a <i>thread local</i> implementation of the
334         * {@code XOR32ShiftRandom} PRNG.
335         *
336         * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
337         * @since 1.0
338         * @version 1.0
339         */
340        public static final class ThreadLocal
341                extends java.lang.ThreadLocal<XOR32ShiftRandom>
342        {
343                private final ParamSelector _selector;
344
345                /**
346                 * Create a new PRNG thread-local instance with the given <em>shift</em>
347                 * and parameter.
348                 *
349                 * @param shift the <em>shift</em> strategy of the PRNG
350                 * @param param the parameter of the PRNG.
351                 * @throws NullPointerException if the given {@code shift} or
352                 *        {@code param} is {@code null}.
353                 */
354                public ThreadLocal(final Shift shift, final Param param) {
355                        _selector = new ParamSelector(
356                                singletonList(shift),
357                                singletonList(param)
358                        );
359                }
360
361                /**
362                 * Create a new PRNG thread-local instance with the given <em>shift</em>
363                 * parameter.
364                 *
365                 * @param shift the <em>shift</em> strategy of the PRNG
366                 * @throws NullPointerException if the given {@code shift} is
367                 *         {@code null}.
368                 */
369                public ThreadLocal(final Shift shift) {
370                        _selector = new ParamSelector(singletonList(shift), Param.PARAMS);
371                }
372
373                /**
374                 * Create a new PRNG thread-local instance with the given parameter.
375                 *
376                 * @param param the parameter of the PRNG.
377                 * @throws NullPointerException if the given {@code param} is
378                 *         {@code null}.
379                 */
380                public ThreadLocal(final Param param) {
381                        _selector = new ParamSelector(
382                                asList(Shift.values()),
383                                singletonList(param)
384                        );
385                }
386
387                /**
388                 * Create a new PRNG thread-local instance.
389                 */
390                public ThreadLocal() {
391                        _selector = new ParamSelector(asList(Shift.values()), Param.PARAMS);
392                }
393
394                /**
395                 * Create a new PRNG using different parameter values for every thread.
396                 */
397                @Override
398                protected XOR32ShiftRandom initialValue() {
399                        final TLRandom random = new TLRandom(
400                                _selector.shift(),
401                                _selector.param(),
402                                _selector.seed()
403                        );
404                        _selector.next();
405
406                        return random;
407                }
408        }

410        /**
411         * Helper class for periodical change or the PRNG parameters <em>shift</em>,
412         * <em>param</em> and <em>seed</em>.
413         */
414        static final class ParamSelector {
415                private final List<Shift> _shifts;
416                private final List<Param> _params;
417                private final int _paramCount;
418
419                private int _index = 0;
420                private byte[] _seed = XOR32ShiftRandom.seedBytes();
421
422                ParamSelector(final List<Shift> shifts, final List<Param> params) {
423                        _shifts = requireNonNull(shifts);
424                        _params = requireNonNull(params);
425                        _paramCount = shifts.size()*params.size();
426                }
427
428                void next() {
429                        if (++_index >= _paramCount) {
430                                _index = 0;
431                                _seed = XOR32ShiftRandom.seedBytes();
432                        }
433                }
434
435                Shift shift() {
436                        return _shifts.get(_index/_params.size());
437                }
438
439                Param param() {
440                        return _params.get(_index%_params.size());
441                }
442
443                byte[] seed() {
444                        return _seed;
445                }
446        }

448        private static final class TLRandom extends XOR32ShiftRandom {
449                private static final long serialVersionUID = 1L;
450
451                private final Boolean _sentry = Boolean.TRUE;
452
453                private TLRandom(
454                        final Shift shift,
455                        final Param param,
456                        final byte[] seed
457                ) {
458                        super(shift, param, seed);
459                }
460
461                @Override
462                public void setSeed(final byte[] seed) {
463                        if (_sentry != null) {
464                                throw new UnsupportedOperationException(
465                                        "The 'setSeed' method is not supported " +
466                                                "for thread local instances."
467                                );
468                        }
469                }
470        }

590        // The largest prime smaller than 2^31.
591        private static final int SAFE_SEED = 2147483647;
592
593        /**
594         * The number of seed bytes (4) this PRNG requires.
595         */
596        public static final int SEED_BYTES = 4;
597
598        private final Shift _shift;
599        private final Param _param;
600
601        private int _x = 0;
602
603        /**
604         * Create a new PRNG instance with the given <em>shift</em>, parameter and
605         * seed.
606         *
607         * @param shift the <em>shift</em> strategy of the PRNG
608         * @param param the parameter of the PRNG.
609         * @param seed the seed of the PRNG.
610         * @throws NullPointerException if the given {@code shift}, {@code param} or
611         *         {@code seed} is {@code null}.
612         * @throws IllegalArgumentException if the given seed is shorter than
613         *         {@link #SEED_BYTES}
614         */
615        public XOR32ShiftRandom(
616                final Shift shift,
617                final Param param,
618                final byte[] seed
619        ) {
620                _shift = requireNonNull(shift, "Shift strategy must not be null.");
621                _param = requireNonNull(param, "PRNG param must not be null.");
622                setSeed(seed);
623        }
624
625        /**
626         * Create a new PRNG instance with the given parameter and seed.
627         *
628         * @param param the parameter of the PRNG.
629         * @param seed the seed of the PRNG.
630         * @throws NullPointerException if the given {@code param} or {@code seed}
631         *         is {@code null}.
632         * @throws IllegalArgumentException if the given seed is shorter than
633         *         {@link #SEED_BYTES}
634         */
635        public XOR32ShiftRandom(final Param param, final byte[] seed) {
636                this(Shift.DEFAULT, param, seed);
637        }
638
639        /**
640         * Create a new PRNG instance with the given parameter and seed.
641         *
642         * @param param the parameter of the PRNG.
643         * @param seed the seed of the PRNG.
644         * @throws NullPointerException if the given {@code param} is {@code null}.
645         */
646        public XOR32ShiftRandom(final Param param, final long seed) {
647                this(param, PRNG.seedBytes(seed, SEED_BYTES));
648        }
649
650        /**
651         * Create a new PRNG instance with the given parameter and a safe seed
652         *
653         * @param param the PRNG parameter.
654         * @throws NullPointerException if the given {@code param} is null.
655         */
656        public XOR32ShiftRandom(final Param param) {
657                this(param, seedBytes());
658        }
659
660        /**
661         * Create a new PRNG instance with the given parameter and seed.
662         *
663         * @param seed the seed of the PRNG.
664         * @throws NullPointerException if the given {@code seed} is {@code null}.
665         * @throws IllegalArgumentException if the given seed is shorter than
666         *         {@link #SEED_BYTES}
667         */
668        public XOR32ShiftRandom(final byte[] seed) {
669                this(Param.DEFAULT, seed);
670        }
671
672        /**
673         * Create a new PRNG instance with {@link Param#DEFAULT} parameter and the
674         * given seed.
675         *
676         * @param seed the seed of the PRNG
677         */
678        public XOR32ShiftRandom(final long seed) {
679                this(Param.DEFAULT, PRNG.seedBytes(seed, SEED_BYTES));
680        }
681
682        /**
683         * Create a new PRNG instance with {@link Param#DEFAULT} parameter and safe
684         * seed.
685         */
686        public XOR32ShiftRandom() {
687                this(Param.DEFAULT, PRNG.seed());
688        }
689
690        /**
691         * Set the seed value of the PRNG.
692         *
693         * @param seed the seed value.
694         * @throws IllegalArgumentException if the given seed is shorter than
695         *         {@link #SEED_BYTES}
696         */
697        public void setSeed(final byte[] seed) {
698                if (seed.length < SEED_BYTES) {
699                        throw new IllegalArgumentException(format(
700                                "Required %d seed bytes, but got %d.",
701                                SEED_BYTES, seed.length
702                        ));
703                }
704
705                _x = toSafeSeed(readInt(seed, 0));
706        }
707
708        private static int toSafeSeed(final int seed) {
709                return seed == 0 ? SAFE_SEED : seed;
710        }
711
712        @Override
713        public void setSeed(final long seed) {
714                _x = toSafeSeed((int)seed);
715        }
716
717        @Override
718        public int nextInt() {
719                return _x = _shift.shift(_x, _param);
720        }
721
722        @Override
723        public String toString() {
724                return String.format("XOR32ShiftRandom[%s, x=%d]", _param, _x);
725        }
726
727        @Override
728        public int hashCode() {
729                int hash = 31;
730                hash += 17*_x + 37;
731                hash += 17*_param.hashCode() + 37;
732
733                return hash;
734        }
735
736        @Override
737        public boolean equals(final Object obj) {
738                return obj instanceof XOR32ShiftRandom &&
739                        ((XOR32ShiftRandom)obj)._x == _x &&
740                        Objects.equals(((XOR32ShiftRandom)obj)._param, _param);
741        }
742
743        /**
744         * Create a new <em>seed</em> byte array suitable for this PRNG. The
745         * returned seed array is {@link #SEED_BYTES} long.
746         *
747         * @see PRNG#seedBytes(int)
748         *
749         * @return a new <em>seed</em> byte array of length {@link #SEED_BYTES}
750         */
751        public static byte[] seedBytes() {
752                return PRNG.seedBytes(SEED_BYTES);
753        }
